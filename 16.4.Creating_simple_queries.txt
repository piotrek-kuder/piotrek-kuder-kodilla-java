-- Ad. 1
SELECT ISSUES.SUMMARY, ISSUELISTS.NAME FROM ISSUES
JOIN ISSUELISTS ON ISSUES.ISSUELIST_ID = ISSUELISTS.ID;

-- Ad. 2
SELECT ISSUES.SUMMARY AS TASK_NAME, USERS.ID AS USER_ID
FROM ISSUES
JOIN USERS ON ISSUES.USER_ID_ASSIGNEDTO = USERS.ID;

-- Ad. 3
SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_AMOUNT
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.ID;
